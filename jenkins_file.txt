pipeline {
    agent any
    stages {
        stage('Checkout') {
            steps {
                // Checkout code from public GitHub repository
                git url: 'https://github.com/Vedhangi/assig9.git'
            }
        }
        stage('Build Docker Image') {
            steps {
                script {
                    // Build Docker image
                    sh 'docker build -t mca2312/assig9 .'
                }
            }
        }
        stage('Run Docker Container') {
            steps {
                script {
                    // Run Docker container in detached mode
                    sh 'docker run -d --name myapp-container mca2312/assig9'
                }
            }
        }
    }
}
